To get the Change request you will make a call to the following endpoint: v1/view/workitem?type=changeRequest

 

And post the following json in the request body

 

[

  {

    "condition": "and",

    "filters": [

      {

        "condition": "and",

        "property": "ScheduledStartDate",

        "operator": "gt",

        "value": "[today]"

      },

      {

        "condition": "and",

        "property": "Downtime",

        "operator": "eq",

        "value": "67ccbc9d-815f-ccef-00ab-3fcabd9e9946"

      },

      {

        "condition": "and",

        "property": "Status",

        "operator": "in",

        "values": [

          "a87c003e-8c19-a25f-f8b2-151b56670e5c",

          "504f294c-ae38-2a65-f395-bff4f085698b",

          "6d6c64dd-07ac-aaf5-f812-6a7cceb5154d",

          "dd6b0870-bcea-1520-993d-9f1337e39d4d"

        ]

      },

      {

        "condition": "and",

        "property": "ScheduledEndDate",

        "operator": "lt",

        "value": "[today]+7d"

      }

    ]

  }

]
